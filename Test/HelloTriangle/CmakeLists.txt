cmake_minimum_required(VERSION 3.10)


set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

project(HelloTriangle VERSION 1.0)

add_executable(HelloTriangle Test.cpp)

target_link_directories(HelloTriangle PRIVATE "${CMAKE_SOURCE_DIR}/Build")
target_link_libraries(HelloTriangle PRIVATE RenderX)
target_include_directories(HelloTriangle PRIVATE "${CMAKE_SOURCE_DIR}/include")
