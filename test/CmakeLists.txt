cmake_minimum_required(VERSION 3.10)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

project(RenderXTest VERSION 1.0)

add_executable(RenderXTest Test.cpp)

target_link_directories(RenderXTest PRIVATE "${CMAKE_SOURCE_DIR}/Build")
target_link_libraries(RenderXTest PRIVATE RenderX)
target_include_directories(RenderXTest PRIVATE "${CMAKE_SOURCE_DIR}/src")
